module LED_8_Display(
        clk ,               // 时钟
        Reset_n,            // 复位
        Seg_EN ,            // 相关数码管使能,也就是Enable
        Seg_Display_LED     // 数码管输出
    );
    // 引脚声明
    input clk ;
    input Reset_n ;
    output [7:0]Seg_EN ;            // 被选中的数码管使能引脚
    output [7:0]Seg_Display_LED;    // 数码管7段 + 1个小数点(默认关闭,暂时保留)
    
    // 库函数参数
    // 计时器
    wire [27:0] cnt ;
    parameter MCNT = 100_000_000 ;   // 20ns一个周期,2s需要2_000_000_000 / 20
    
    // 数码管数据显示
    reg LED_8_Flag ;                    // 数码管全体开关
    reg [2:0] Seg_Num ;                 // 数码管编号选择
    reg [3:0] Seg_Display_Num ;         // 数码管需要显示的数据
    
    // 8段数码管逻辑
    always @(posedge clk or negedge Reset_n)
        if (!Reset_n) begin
            LED_8_Flag <= 0 ;   // 初始时不显示LED
            Seg_Num <= 3'd0 ;
            Seg_Display_Num <= 4'd0 ;
        end 
        else if (cnt == MCNT / 4 - 1) begin 
            Seg_Num <= 3'd1 ;
            Seg_Display_Num <= 4'h8 ;
        end
        else if(cnt == MCNT / 2 - 1) begin 
            Seg_Num <= 3'd5 ;
            Seg_Display_Num <= 4'hf ;
        end
        else if(cnt == MCNT  - 1) begin 
            Seg_Num <= 3'd0 ;
            Seg_Display_Num <= 4'h0 ;
        end
        else 
            LED_8_Flag <= 1;    // 使能就开
    
    // 数码管展示
    LED_1_Display LED_1_Display_1(
        .Seg_Enable(LED_8_Flag) ,                // 启动引脚,和Reset_n一样的
        .Seg_Num(Seg_Num)  ,                    // 数码管的编号,一共8个,所以是三位0-7
        .Seg_Display_Num(Seg_Display_Num)  ,    // 数码管需要展示的字符
        .Seg_EN(Seg_EN)  ,                      // 相关数码管使能,也就是Enable
        .Seg_Display_LED(Seg_Display_LED)       // 数码管输出
    ) ;
    
    // 计时器
    Timer Timer_1(
        .Reset_n(Reset_n) ,   // 复位
        .clk(clk) ,       // 时钟
        .cnt(cnt) ,       // 计时
        .MCNT(MCNT)        // 最大计时
    ) ;
endmodule
